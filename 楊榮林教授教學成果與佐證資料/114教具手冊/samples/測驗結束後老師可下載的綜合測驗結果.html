const rawPacket = `[
  {"id":"B01","time":"09:00","voltage":12.6,"current":0.5},
  {"id":"B01","time":"09:10","voltage":11.4,"current":0.6},
  {"id":"B01","time":"09:20","voltage":12.2,"current":0.4},

]`;

  }
